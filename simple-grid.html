CSS


.grid {
    display: grid;
}

/* Adding this class will set our grid to have exactly 1 column */
.grid-cols-1 {
    grid-template-columns: repeat(1, minmax(0, 1fr));
}

/* 
 * On larger screens we want our grid to accomodate 2 columns of images.
 * We then use `@media` to specify the minimum breakpoint where we want to start
 * displaying 2 columns. In this case if the viewport is at least at 768px wide.
 */
@media (min-width: 768px) {
    .md\:grid-cols-2 {
        grid-template-columns: repeat(2, minmax(0, 1fr));
    }
}

/* Adding this on an element will make its height expand up to the height of its container */
.h-full {
    height: 100%;
}

/* Resize an elementâ€™s content to cover its container using, perfect for responsive images */
.object-cover {
    object-fit: cover;
}
